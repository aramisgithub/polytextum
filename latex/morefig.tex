%
% morefig.sty -- Figure setup
%

% Allow custom captions
\RequirePackage{caption}

% Inline figure caption settings
\captionsetup{
  font={small},
  justification=RaggedRight,
  singlelinecheck=false
}

% Custom inline figure
\newcommand{\myinfig}[2]{
  \begin{figure}[ht]
    \vspace{0bp plus 8bp}
    \centering
    \includegraphics[width=\linewidth]{#1}
    \caption*{#2}
    \vspace{0bp plus 8bp}
  \end{figure}
}
