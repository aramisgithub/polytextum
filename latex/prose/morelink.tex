% morelink.sty -- Hyperref

% Add PDF hyperlinks (but don't link footnotes!)
\RequirePackage[hyperfootnotes=false]{hyperref}

% Sanserif links
\urlstyle{sf}

% Set link colours
\hypersetup{%
  colorlinks=true,
  linkcolor=black,
  anchorcolor=black,
  citecolor=black,
  urlcolor=brown,
  breaklinks,
  pdfborder=0 0 0,
}
