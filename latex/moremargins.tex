%
% moremargins.sty -- Margin-related settings
%

% If sidenotes are enabled ...
\ifthenelse{\boolean{@argon@sidenotes}}{%
  % Set width of typeblock for a line length of ~66 characters, and set the
  % typeblock height to be equal to 40 lines.
  \settypeblocksize{40\baselineskip}{26pc}{*}

  % Use a more extreme horizontal margin ratio to make space for the sidenotes
  \setlrmargins{*}{*}{3}

  % Set up the space the sidenotes will occupy
  \setmarginnotes{2pc}{12pc}{\baselineskip}

  % And set all the sidenotes settings
  \input{moresidenotes}

  % If not using sidenotes ...
}{

  % Set width of typeblock for a line length of ~72 characters, and set the
  % typeblock height to be equal to 40 lines. Although 26pc and 66 cpm would be
  % ideal, having such wide margins looks a little odd.
  \settypeblocksize{40\baselineskip}{29pc}{*}

  % Use asymmetrical margins if using a two-sided layout. Otherwise set the left
  % and right margins to be equal
  \ifthenelse{\boolean{@argon@twoside}}{%
    \setlrmargins{*}{*}{2}
  }{
    \setlrmargins{*}{*}{1}
  }

  % Place all figures inline with the text
  \newcommand{\myfig}[2]{
    \begin{figure}[ht]
      \vspace{0bp plus 8bp}
      \centering
      \includegraphics[width=\linewidth]{#1}
      \caption*{#2}
      \vspace{0bp plus 8bp}
    \end{figure}
  }
}

% Traditional vertical margin ratio
\setulmargins{*}{*}{2}

\checkandfixthelayout
