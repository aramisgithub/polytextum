% moretitling.sty -- Titles and titlepages

% Typeset an ordinary title with \maketitle
\renewcommand{\maketitle}{
  \normalfont

  % Title in small caps
  {\stoppar\scshape{\thetitle}\par}
  % Author name and date in italics, dark grey to deemphasize
  {\stoppar\color{subtitle}\itshape\theauthor\\\thedate\par}

  % One empty line
  \vspace{\baselineskip}

  % This removes the indent that may be present after the \maketitle
  \@afterindentfalse
  \@afterheading

  \normalfont
}

% Typeset a titlepage with \maketitlepage
\newcommand{\maketitlepage}[2]{
  \begin{titlingpage}
    \null\vspace{8pc}

    {\large\theauthor\par}
    \vspace{3pc}

    {\Large\color{bigtitle}\uppercaps{#1}\par}
    \vspace{0.5em}
    {\Large\color{bigtitle}\uppercaps{#2}\par}
    \vspace{3pc}

    \vfill
    {\large \thedate\par}
  \end{titlingpage}
}
