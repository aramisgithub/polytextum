% moretitling.sty -- Titles and titlepages

% Typeset an ordinary title with \maketitle
\renewcommand{\maketitle}{
  \normalfont

  % Title in small caps
  {\stoppar\scshape{\thetitle}\par}
  % Author name and date in italics, dark grey to deemphasize
  {\stoppar\color{subtitle}\itshape\theauthor\\\thedate\par}

  % One empty line
  \vspace{\baselineskip}

  % This removes the indent that may be present after the \maketitle
  \@afterindentfalse
  \@afterheading

  \normalfont
}

% Typeset a titlepage with \maketitlepage
\newcommand{\maketitlepage}[4]{
  \begin{titlingpage}
    \null\vspace{8pc}

    {\stoppar\large#1\par}
    {\stoppar\large#2\par}
    \vspace{3pc}

    {\stoppar\Large\color{bigtitle}\uppercaps{#3}\par}
    {\stoppar\Large\color{bigtitle}\uppercaps{#4}\par}
    \vspace{3pc}

    \vfill
    {\stoppar\large \thedate\par}
  \end{titlingpage}
}
